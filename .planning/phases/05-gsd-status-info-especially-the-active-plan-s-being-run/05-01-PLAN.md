```python
def _run_git(args: list[str], cwd: str, timeout: float = 0.15) -> str | None:
    try:
        proc = subprocess.run(["git", "-C", cwd, *args], capture_output=True, text=True, timeout=timeout)
        if proc.returncode != 0:
            return None
        return proc.stdout
    except Exception:
        return None
```
